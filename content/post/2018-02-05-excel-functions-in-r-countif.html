Excel Syntax:

COUNTIF(range, criteria)

R Equivalent:

library(dplyr)

DataFrame$count <- split(DataFrame, DataFrame["range"]) %>% 
 lapply(nrow) %>%
 unsplit(DataFrame["range"])
